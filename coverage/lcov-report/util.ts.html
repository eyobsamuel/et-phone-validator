}
 
export function extractNumber(phone: string) {
  return phone.split(" ")
    .join("")
    .split("-")
    .join("")
    .split("(")
    .join("")
    .split(")")
    .join("");
}

export function checkEthioTelecomFixedLine(phone: any, patterns: any) {
  let found = false
  patterns.forEach((pattern: any) => {
    if (pattern.test(phone))
      found = true
    else {
      // do-nothing
    }
  });
  if (found)
    return true
  else
    return false
}
